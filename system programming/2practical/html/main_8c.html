{

str[0] = '\0';
int lenght = 1;
char cur_char = 0;
fflush(stdin);
while ((cur_char = fgetc(file)) != EOF)
{
str[lenght - 1] = cur_char;
lenght++;
str = (char *)realloc(str, lenght);
}
str[lenght - 1] = '\0';
return str;
}
